scripted_map_modes = {
	market_spheres_mapmode = {
		top = {
			type = country

			color = {
				FROM = {
					if = {
						limit = { has_idea = sphere_leader }
						set_temp_variable = { red = global.red@var:FROM }
						set_temp_variable = { green = global.green@var:FROM }
						set_temp_variable = { blue = global.blue@var:FROM }
						set_temp_variable = { alpha = 1 }
					}
					else_if = {
						limit = {
							OR = {
								has_idea = sphere_member_level_1
								has_idea = sphere_member_level_2
								has_idea = sphere_member_level_3
							}
						}
						set_temp_variable = { red = global.red@var:sphere_overlord }
						set_temp_variable = { green = global.green@var:sphere_overlord }
						set_temp_variable = { blue = global.blue@var:sphere_overlord }
						set_temp_variable = { alpha = 0.65 }
					}
					else = {
						set_temp_variable = { red = 0.5 }
						set_temp_variable = { green = 0.5 }
						set_temp_variable = { blue = 0.5 }
						set_temp_variable = { alpha = 1 }
					}
				}
			}

			thickness = yes

			targets = {

			}
		}
		bottom = {
			type = country
			color = {
				set_temp_variable = {
					alpha = 0.1
				}
				set_temp_variable = {
					thickness = 1
				}
				set_temp_variable = {
					red = 0.33
				}
				set_temp_variable = {
					green = 0.33
				}
				set_temp_variable = {
					blue = 0.33
				}
			}
			thickness = yes
		}

		far_text = country
		near_text = country

		update_daily = no
	}
}
